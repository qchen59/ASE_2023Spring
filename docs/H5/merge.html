def merge(col1, col2):
    isNew = deepcopy(col1)
    if isinstance(col1, sym.Sym):
        for x, n in col2.has.items():
            isNew.add(x,n)
    else:
        for n in col2.has:
            isNew.add(n)
        isNew.lo = min(col1.lo, col2.lo)
        isNew.hi = max(col1.hi, col2.hi)
    return isNew
